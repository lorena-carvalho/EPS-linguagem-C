EP 1.3 - Conta números primos
```
#include <stdio.h>

int main(void) {
	int min, max, i, j, primo, cont=0;
	scanf("%i %i", &min, &max);
	
	for (i = min; i <= max; i++) {
		
    if (i == 0 || i == 1) {
			primo = 0;
      
		} else {
			primo = 1; 
			for(j = 2; j < i; j++) { 
				if (i % j == 0) {
					primo = 0; 
				}
			}
		}
		if (primo == 1) {
      cont++;    
		}
	}
  printf("%d\n",cont);
	
	return 0;
}
```

EP 1.4 - Ordem crescente
```
#include <stdio.h>
 
int main(){
  int n,i,num,ant,crescente = 1;
  
  scanf("%d",&n);
 
  scanf("%d",&ant);
  i = 2;
  while(i <= n && crescente){
    scanf("%d",&num);
    if(num <= ant)
      crescente = 0;
    ant = num;
    i = i + 1;
  }
  if(crescente)
    printf("SIM\n");
  else
    printf("NAO\n");
 
  return 0;
}
```
